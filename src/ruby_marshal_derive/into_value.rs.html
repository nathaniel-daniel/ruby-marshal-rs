13pub fn derive(input: proc_macro::TokenStream) -> proc_macro::TokenStream {
14    let input = parse_macro_input!(input as DeriveInput);
15    let input_data = match &input.data {
16        syn::Data::Struct(data) => data,
17        _ => {
18            return quote_spanned! {
19                input.span() =>
20                compile_error!("only structs are supported");
21            }
22            .into();
23        }
24    };
25    let input_fields = match &input_data.fields {
26        syn::Fields::Named(fields) => fields,
27        _ => {
28            return quote_spanned! {
29                input.span() =>
30                compile_error!("only named field structs are supported");
31            }
32            .into();
33        }
34    };
35
36    let container_attributes =
37        match parse_container_attributes(&input).map_err(syn::Error::into_compile_error) {
38            Ok(value) => value,
39            Err(error) => {
40                return error.into();
41            }
42        };
43    let object_name = container_attributes;
44
45    let mut fields = Vec::with_capacity(input_fields.named.len());
46    for field in input_fields.named.iter() {
47        let field_attributes =
48            match parse_field_attributes(field).map_err(syn::Error::into_compile_error) {
49                Ok(value) => value,
50                Err(error) => {
51                    return error.into();
52                }
53            };
54
55        let name = field
56            .ident
57            .as_ref()
58            .expect("named field structs should have named fields");
59
60        let name_str = match field_attributes.name {
61            Some(name) => name,
62            None => LitByteStr::new(format!("@{name}").as_bytes(), name.span()),
63        };
64        fields.push(IntoValueField {
65            name,
66            name_str,
67            ty: &field.ty,
68            into_value: field_attributes.into_value,
69        });
70    }
71
72    let create_field_keys = fields.iter().enumerate().map(|(i, field)| {
73        let ident = format_ident!("field_{i}_key");
74        let name = &field.name_str;
75        quote! {
76            let #ident = arena.create_symbol(#name.into());;
77        }
78    });
79
80    let create_field_values = fields.iter().enumerate().map(|(i, field)| {
81        let ident = format_ident!("field_{i}_value");
82        let field_name = &field.name;
83        let ty = &field.ty;
84        let ty_span = ty.span();
85
86        match field.into_value.as_ref() {
87            Some(into_value) => {
88                quote_spanned! {into_value.span()=>
89                    let #ident = #into_value(self.#field_name, arena)?;
90                }
91            }
92            None => {
93                let cast_type = quote_spanned! {ty_span=>
94                    <#ty as ::ruby_marshal::IntoValue>
95                };
96                quote! {
97                    let #ident = #cast_type::into_value(self.#field_name, arena)?;
98                }
99            }
100        }
101    });
102
103    let field_vec_entries = fields.iter().enumerate().map(|(i, _field)| {
104        let key_ident = format_ident!("field_{i}_key");
105        let value_ident = format_ident!("field_{i}_value");
106
107        quote! {
108            (#key_ident, #value_ident),
109        }
110    });
111
112    let input_name = &input.ident;
113    let tokens = quote! {
114        impl ::ruby_marshal::IntoValue for #input_name {
115            fn into_value(
116                self,
117                arena: &mut ::ruby_marshal::ValueArena
118            ) -> Result<::ruby_marshal::ValueHandle, ::ruby_marshal::IntoValueError> {
119                let object_name = arena.create_symbol(#object_name.into());
120
121                #(#create_field_keys)*
122
123                #(#create_field_values)*
124
125                let fields = vec![
126                    #(#field_vec_entries)*
127                ];
128
129                let object = arena.create_object(object_name, fields);
130
131                Ok(object.into())
132            }
133        }
134    };
135
136    proc_macro::TokenStream::from(tokens)
137}
138
139struct IntoValueField<'a> {
140    name: &'a Ident,
141    name_str: LitByteStr,
142    ty: &'a Type,
143    into_value: Option<syn::Path>,
144}
